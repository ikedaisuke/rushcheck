  def self.arbitrary
    RushCheck::Gen.new do |n, r|
      Proc.new do |*args|
        gens = if args.empty?
               then @@outputs.map {|c| c.arbitrary }
               else args.map do |v|
                      @@outputs.map do |c|
                        v.coarbitrary(c.arbitrary)
                      end
                    end.flatten
               end
        RushCheck::Gen.oneof(gens).value(n, r)
      end
    end
  end
